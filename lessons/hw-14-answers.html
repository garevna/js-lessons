const typeMessage = async ( message, container ) => {
    while ( message.length ) {
        var [ letter, message ] = [ message.slice ( 0, 1 ), message.slice ( 1 ) ];
        container.textContent += await new Promise (
            resolve => setTimeout ( () => resolve ( letter ), 1000 )
        )
    }
}

typeMessage (
     "Hello, students!",
    document.body.appendChild (
        document.createElement ( "p" )
    )
)

const typeMessage = async ( message, container ) => {
    var [ letter, message ] = [ message.slice ( 0, 1 ), message.slice ( 1 ) ];
    container.textContent += await new Promise (
        resolve => setTimeout ( () => resolve ( letter ), 1000 )
    )
    return message.length && typeMessage ( message, container )
}
